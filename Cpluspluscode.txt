#include <iostream>
using namespace std;

float factorialN(int n){
    if (n == 1 || n == 0){
        return 1;
    }
    else{
        return(n * factorialN(n-1));
    }
}

float XpowerN(float x,int n){
    if (n == 0){
        return 1;
    }
    else{
        return(x * XpowerN(x,n-1));
    }
}

float powerpertorial(float x, int n){
    return (XpowerN(x,n)/factorialN(n));
}
